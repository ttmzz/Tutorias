Código Gulp + Sass


// Importa os pacotes
const gulp = require('gulp');
const sass = require('gulp-sass')(require('sass'));
const sourcemaps = require('gulp-sourcemaps');
const cleanCSS = require('gulp-clean-css');
const uglify = require('gulp-uglify');
const obfuscate = require('gulp-obfuscate'); // faz com que proteja as informações no javascript
const imagemin = require('gulp-imagemin');


// Função para comprimir Imagens
function comprimeImagens() {
    return gulp.src('./source/imagens/*')
    .pipe(imagemin())
    .pipe(gulp.dest('./build/imagens'));
}

// Função para comprimir o JavaScript
function comprimeJavascript() {
    return gulp.src('./source/scripts/*.js')
    .pipe(uglify())
    .pipe(obfuscate())
    .pipe(gulp.dest('./build/scripts'))
}            

// Função para compilar o Sass
function compilaSass() {
    return gulp.src('./source/styles/main.scss')
    .pipe(sourcemaps.init())
    .pipe(sass().on('error', sass.logError))
    .pipe(cleanCSS()) // Minifica o CSS gerado
    .pipe(sourcemaps.write('./maps'))
    .pipe(gulp.dest('./build/styles'));
}


// Tarefas exportadas
exports.default = function() {
    gulp.watch('./source/styles/*.scss', { ignoreInitial: false}, gulp.series(compilaSass));
    gulp.watch('./source/scripts/*.js', { ignoreInitial: false}, gulp.series(comprimeJavascript));
    gulp.watch('./source/imagens/*', { ignoreInitial: false}, gulp.series(comprimeImagens));
}


OBS: Para instalar e rodar o gulp-clean-css é preciso abrir o terminal e usar o comando ->  npm install gulp-clean-css --save-dev
